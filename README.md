To Install Copy The BanWhiteList Folder to DiscordBot folder.

    'BanWhiteList/BanWhitelistConfig.lua',
    'BanWhiteList/Server.lua'
}
files {
    'BanWhiteList/Whitelists.json',
    'BanWhiteList/Bans.json'
}
